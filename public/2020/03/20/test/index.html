年次

リターン

data_year <- data %>%
  dplyr::group_by(year(date)) %>%
  do(head(., 1))
data_year <- data_year %>%
  ungroup(data_year) %>%
  dplyr::mutate(lag=(close-dplyr::lag(close))/dplyr::lag(close))

datatable(data_year,
        filter = 'top', 
        style = 'bootstrap', class = 'table-bordered table-condensed',
        extensions = 'ColReorder',
        options = list(dom = 'Rlfrtip')
        )
